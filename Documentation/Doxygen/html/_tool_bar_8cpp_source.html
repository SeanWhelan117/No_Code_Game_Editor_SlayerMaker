409void ToolBar::pulseTriangles()
410{
411
412 if (scalingUp && navigationTriangles[0].getScale().x < maxScale.x)
413 {
414 for (int i = 0; i < MAX_NAV_TRIANGLES; i++)

420 {

428 {
429 scalingUp = true;
